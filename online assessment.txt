<script>
  function calculateScore() {
    let score = 0;
    if (document.getElementById("q1").value == 8) score++;
    if (document.getElementById("q2").value == 4) score++;
    if (document.getElementById("q3").value == 8) score++;
    if (document.getElementById("q4").value == 8) score++;
    if (document.getElementById("q5").value == 4) score++;

    // Hide quiz, show thank you
    document.getElementById("quizContainer").style.display = "none";
    document.getElementById("thankYouPage").style.display = "block";

    // Show score
    document.getElementById("scoreBox").innerText = "Your Score: " + score + " / 5";
  }
</script>
